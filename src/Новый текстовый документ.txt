#include <TFT_eSPI.h>
TFT_eSPI tft = TFT_eSPI();
TFT_eSprite spr = TFT_eSprite();

void setup() {
  tft.init();
  tft.setRotation(1);
  tft.fillScreen(TFT_BLACK);
  
  // Инициализируем спрайт
  spr.createSprite(tft.width(), tft.height());
  spr.fillSprite(TFT_BLACK);
  
  // Загружаем шрифт
  spr.loadFont("timesNewRoman24");
  spr.setTextColor(TFT_ORANGE, TFT_BLACK);
  
  // Рисуем текст в спрайт
  spr.setCursor(0, 70);
  spr.println("Привет Мир!");
  
  // Плавный выезд спрайта
  for(int x = -spr.width(); x < tft.width(); x++) {
    tft.fillScreen(TFT_BLACK);
    spr.pushSprite(x, 0);
    delay(20);
  }
  
  // Очистка ресурсов
  spr.deleteSprite();
  spr.unloadFont();
}

void loop() {
  // Ничего не делаем
}


Преимущества использования спрайтов:

    Более плавная анимация без мерцания

    Меньшая нагрузка на процессор

    Возможность более сложных манипуляций с текстом

    Быстрее отрисовка при повторном использовании

Для создания эффекта затухания при въезде можно модифицировать код:

// Плавный выезд с затуханием
for(int x = -spr.width(); x < tft.width(); x++) {
  tft.fillScreen(TFT_BLACK);
  spr.pushSprite(x, 0);
  
  // Вычисляем коэффициент прозрачности
  float alpha = constrain(map(x, 0, tft.width(), 0, 255), 0, 255);
  tft.setBlend(alpha);
  tft.pushImage(x, 0, spr.width(), spr.height(), spr.getBuffer());
  
  delay(20);
}

Дополнительные возможности:

    Создание циклического движения

    Добавление вращения спрайта

    Комбинирование нескольких спрайтов

    Использование различных эффектов перехода

Для создания циклического движения можно использовать:

// Бесконечный цикл движения
while(true) {
  for(int x = -spr.width(); x < tft.width(); x++) {
    tft.fillScreen(TFT_BLACK);
    spr.pushSprite(x, 0);
    delay(20);
  }
  delay(1000); // Пауза перед повторением
}

Такой подход позволяет создавать более сложные и интересные визуальные эффекты на дисплее.
